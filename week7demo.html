outer_labels =cars_data['cyl'].value_counts().index
outer_sizes = cars_data['cyl'].value_counts().values

inner_labels = cars_data['gear'].value_counts().index
inner_sizes = cars_data['gear'].value_counts().values

# Create nested pie chart
fig, ax = plt.subplots(figsize=(8, 8))
ax.pie(outer_sizes, labels=outer_labels, radius=1, autopct='%1.1f%%', colors=['skyblue', 'lightgreen', 'orange'])
ax.pie(inner_sizes, labels=inner_labels, radius=0.7, autopct='%1.1f%%', colors=['pink', 'purple', 'cyan'])
ax.set(aspect="equal")
plt.title('Nested Pie Chart: Cylinders (Outer) and Gears (Inner)')
plt.show()
